Gob's Program in Haskell

Thursday, 17 March 2011 at 10:05 AM

(via Myles Braithwaite) 

import System.Exit
import Control.Monad
main = do
	putStrLn "Gob's Program: Y/N?"
	ans <- getLine
	case ans of
		"Y" -> sequence_ $ map putStr penuses
		_   -> exitSuccess

penuses = ["Penus " | _ <- [1..]]
